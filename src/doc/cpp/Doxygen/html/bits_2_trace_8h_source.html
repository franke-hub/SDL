34namespace _PUB_NAMESPACE::detail::Trace { // Detail namespace for Trace.cpp
35//----------------------------------------------------------------------------
36// Special Case Debug Mode-
37// SCDM > 0 enables the special case logic in allocate(), which counts the
38// compare_exchange retries in spins.
39// When spins > SCDM, nullptr is returned to the caller, and we replace the
40// allocated record a .TAF (Trace Allocation Failure) record. Additionally,
41// if USE_DEACTIVATE, we invoke deactivate() (to terminate tracing.)
42// (USE_DEACTIVATE has no effect unless SCDM is enabled.)
43//----------------------------------------------------------------------------
44enum // Compile-time options
45{ CHECK= false // Check for should not occur conditions?
46, HCDM= false // Hard Core Debug Mode?
47, SCDM= 0 // Special Case Debug Mode, spin limit
48, USE_DEACTIVATE= false // SCDM(Deactivate trace option)
49}; // Compile-time options
50} // namespace _PUB_NAMESPACE::detail::Trace
51#endif // _PUB_BITS_TRACE_H_INCLUDED
