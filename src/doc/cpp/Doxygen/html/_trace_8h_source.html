12// Title-
13// Trace.h
14//
15// Purpose-
16// Trace table storage allocator.
17//
18// Last change date-
19// 2022/03/11
20//
21// Usage notes-
22// The Trace object allocates storage sequentially from itself, wrapping
23// when a request cannot otherwise be satisfied. This is intended for use
24// as a trace table item allocator. Applications may find other uses.
25//
26// The Trace object is thread-safe and process-safe. Trace tables may be
27// allocated in shared memory and shared between processes. Process's
28// Trace::table may differ.
29//
30// A trace table SHOULD be large enough that so that table wrapping does
31// not cause initialization interference.
32//
33// Implementation notes-
34// Applications are responsible for trace table allocation/release.
35// The Trace object is contained within the trace table. The entire
36// trace table is initialized using make().
37//
38// Trace storage allocation is performance critical. Therefore, this
39// methods DOES NOT normally check for application errors. There is
40// NO defined application mechanism to enable this checking.
41//
42// Specifically, Trace::allocate MIGHT NOT check whether:
43// * flag[X_HALT] == 0 (allocate_if provides this check)
44// * Parameter size == 0 (Always an application error)
45// * Parameter size > (this->size - sizeof(Trace))
46// * Possible arithemetic overflow: size > (size + this->next)
47// This condition can occur with large trace tables and large
48// allocation requests. To prevent this, applications MUST NOT
49// allow the size parameter plus the size of the trace table to
50// exceed UINT32_MAX.
51//
52// Sample usage-
53// void* storage= malloc(desired_size); // Unaligned is OK. make trims
54// Trace::table= pub::Trace::make(storage, desired_size);
55//
56// struct Record : public pub::Trace::Record { // Your Record
57// : // Your data goes here
58// };
59// :
60// Record* record= (Record*)Trace::storage_if(sizeof(Record));
61// if( record ) { // If trace is active
62// : // Initialize your data
63// record->trace(".xxx"); // Initialize the trace identifier + clock
64// }
65// :
66// free(Trace::table); Trace::table= nullptr; // Done with trace table
67//

91//----------------------------------------------------------------------------
92// Trace::Buffer, temporary character string storage area
93//----------------------------------------------------------------------------
94template<size_t N> // Buffer.temp *always* fully used
95struct Buffer {
96char temp[N]; // The temporary Buffer, '\0' padded
97
98 Buffer(const void* info, size_t size)
99{ if( size < N ) {
100 memcpy(temp, info, size);
101 memset(temp+size, '\0', N-size);
102 } else {
103 memcpy(temp, info, N);
104 }
105}
106
107 Buffer(const void* info)
108{ size_t size= strlen((const char*)info);
109 if( size < N ) {
110 memcpy(temp, info, size);
111 memset(temp+size, '\0', N-size);
112 } else {
113 memcpy(temp, info, N);
114 }
115}
116}; // struct Buffer

118//----------------------------------------------------------------------------
119// Trace::Record (POD: Plain Old Data)
120//----------------------------------------------------------------------------
121struct Record { // A standard (POD) trace record
122char ident[4]; // The trace type identifier
123uint32_t unit; // The trace unit identifier
124uint64_t clock; // The UTC epoch clock, in nanoseconds
125char value[16]; // Data values (For smallest Record)
126
127inline void
128 set_clock() // Set the clock
129{
130 struct timespec clock; // UTC time base
131 clock_gettime(CLOCK_REALTIME, &clock); // Get UTC time base
132#if true // Use big_endian clock value?
133 uint64_t nsec= (clock.tv_sec << 32) | clock.tv_nsec;
134 char* addr= (char*)&this->clock; // clock.tv_nsec: 0x00000000..0x3B9AC9FF
135 for(int i= 8; i>0; i--) {
136 addr[i-1]= char(nsec);
137 nsec >>= 8;
138 }
139#else
140 record->clock= (clock.tv_sec * 1000000000) + clock.tv_nsec;
141#endif
142}
143
144inline void
145 trace( // Initialize with
146 const char* ident) // This char[4] trace type identifier
147{ set_clock(); // Set the clock
148 memcpy(this->ident, ident, sizeof(this->ident));
149}
150
151inline void
152 trace( // Initialize with
153 const char* ident, // This char[4] trace type identifier
154 uint32_t code, // Trace code
155 const void* info= nullptr) // If present, char[16] info
156{ unit= htonl(code);
157 if( info ) {
158 memcpy(value, info, sizeof(value));
159 }
160 trace(ident);
161}
162
163inline void
164 trace( // Initialize with
165 const char* ident, // This char[4] trace type identifier
166 const char* unit, // This char[4] trace subtype identifier
167 void* one= nullptr, // Word one
168 void* two= nullptr) // Word two
169{ memcpy(&this->unit, unit, sizeof(this->unit));
170 uintptr_t uone= uintptr_t(one);
171 uintptr_t utwo= uintptr_t(two);
172 for(unsigned i= 8; i>0; i--) {
173 value[i-1]= char(uone);
174 value[i+7]= char(utwo);
175 uone >>= 8;
176 utwo >>= 8;
177 }
178 trace(ident);
179}
180}; // struct Record
