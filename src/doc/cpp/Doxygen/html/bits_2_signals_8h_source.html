30//----------------------------------------------------------------------------
31//
32// Class-
33// pub::signals::__detail::Listener
34//
35// Purpose-
36// Listener descriptor, contains a std::function<void(Event&)> object.
37//
38// Implementation notes-
39// In ListenerList, this class is aliased as a Slot_t.
40//
41//----------------------------------------------------------------------------
42template<typename Event>
43class Listener : public pub::List<Listener<Event>>::Link { // Listener descriptor
44//----------------------------------------------------------------------------
45// pub::Listener::Attribute
46//----------------------------------------------------------------------------
47protected:
48typedef std::function<void(Event&)>
49 Function; // Function<Event> class
50
51const Function function; // The Event handler function
52
53//----------------------------------------------------------------------------
54// pub::signals::__detail::Listener::Constructor
55//----------------------------------------------------------------------------
56public:
57 Listener( // Constructor
58 const Function& _function) // The Event handler function
59: function(_function)
60{ if( pub_hcdm )
61 debugf("Listener(%p.%zd)::Listener(%p.%zd)\n"
62 , this, sizeof(*this), &_function, sizeof(_function));
63}
64
65//----------------------------------------------------------------------------
66// pub::signals::__detail::Listener::Destructor
67//----------------------------------------------------------------------------
68 ~Listener( void ) // Destructor
69{ if( pub_hcdm ) debugf("Listener(%p)::~Listener\n", this); }
70
71//----------------------------------------------------------------------------
72// pub::signals::__detail::Listener::signal
73//----------------------------------------------------------------------------
74void
75 signal( // Signal this Listener about
76 Event& event) const // This (application defined) Event
77{ if( pub_hcdm ) debugf("Listener(%p)::signal(%p)\n", this, &event);
78
79 function(event);
80}
81}; // class Listener

83//----------------------------------------------------------------------------
84//
85// Class-
86// pub::signals::__detail::ListenerList
87//
88// Purpose-
89// The List<Listener<Event>> container, with locking controls
90//
91//----------------------------------------------------------------------------
92template<typename Event>
93class ListenerList { // The List of Listeners (container)
94//----------------------------------------------------------------------------
95// pub::signals::__detail::ListenerList::Attributes
96//----------------------------------------------------------------------------
97protected:
98typedef Listener<Event>Slot_t; // The Listener class alias
99
100mutable
101::pub::SharedLatch SHR; // Protects the List of Listeners
102::pub::List<Slot_t> list; // The actual List of Listeners
103
104//----------------------------------------------------------------------------
105// pub::signals::__detail::ListenerList::Constructor
106//----------------------------------------------------------------------------
107public:
108 ListenerList( void ) // Default constructor
109: SHR(), list()
110{ if( pub_hcdm ) debugf("ListenerList(%p)::ListenerList\n", this); }
111
112//----------------------------------------------------------------------------
113// pub::signals::__detail::ListenerList::Destructor
114//----------------------------------------------------------------------------
115 ~ListenerList( void ) // Destructor
116{ if( pub_hcdm ) debugf("ListenerList(%p)::~ListenerList\n", this);
117
118 // Implementation note: The ListenerList may contain Listeners. Only the
119 // Connector may delete these objects and it can't find the ListenerList.
120 // We, on the other hand, can no longer access the ListenerList because it
121 // may contain deleted elements. So, no action is wanted here.
122}
123
124//----------------------------------------------------------------------------
125// pub::signals::__detail::ListenerList::debug
126//----------------------------------------------------------------------------
127void
128 debug( void ) // Debugging display
129{
130 size_t X= 0;
131 std::lock_guard<decltype(SHR)> lock(SHR); // While holding the shared Latch
132 for(Slot_t* slot= list.get_head(); slot; slot= slot->get_next() ) {
133 debugf("[%2zd] %p\n", X++, slot); // (Note that X is incremented here)
134 }
135}
136
137//----------------------------------------------------------------------------
138//
139// Method-
140// pub::signals::__detail::ListenerList::signal
141//
142// Purpose-
143// Signal Event occurance
144//
145//----------------------------------------------------------------------------
146void // (All Listeners are signaled)
147 signal( // Signal all Listeners about
148 Event& event) const // This Event
149{ if( pub_hcdm ) debugf("ListenerList(%p)::signal(%p)\n", this, &event);
150
151 std::lock_guard<decltype(SHR)> lock(SHR); // While holding the shared Latch
152 for(Slot_t* slot= list.get_head(); slot; slot= slot->get_next())
153 slot->signal(event); // signal the Listener
154}
155
156//----------------------------------------------------------------------------
157// pub::signals::__detail::ListenerList::insert
158//----------------------------------------------------------------------------
159void
160 insert( // Insert
161 Slot_t* slot) // This Listener (FIFO ordering)
162{ if( pub_hcdm ) debugf("ListenerList(%p)::insert(%p)\n", this, slot);
163
164 ::pub::ExclusiveLatch XCL(SHR); // While holding the exclusive Latch
165 std::lock_guard<decltype(XCL)> lock(XCL);
166
167 list.fifo(slot); // Insert the Slot
168}
169
170//----------------------------------------------------------------------------
171// pub::signals::__detail::ListenerList::remove
172//----------------------------------------------------------------------------
173void
174 remove( // Remove
175 Slot_t* slot) // This Listener
176{ if( pub_hcdm ) debugf("ListenerList(%p)::remove(%p)\n", this, slot);
177
178 ::pub::ExclusiveLatch XCL(SHR); // While holding the exclusive Latch
179 std::lock_guard<decltype(XCL)> lock(XCL);
180
181 list.remove(slot, slot); // Remove the Listener
182}
183}; // class ListenerList
