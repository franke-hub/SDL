31#include "config.h" // For _PUB_NAMESPACE

33namespace _PUB_NAMESPACE {
34//----------------------------------------------------------------------------
35//
36// Struct-
37// pub::Statistic
38//
39// Purpose-
40// Statistic counter.
41//
42// Implementation note-
43// Use current.load() to get current value, etc. for maximum/minimum.
44//
45//----------------------------------------------------------------------------
46struct Statistic { // Statistic
47std::atomic_int64_t counter= 0; // Total increment count
48std::atomic_int64_t current= 0; // Current value
49std::atomic_int64_t maximum= 0; // Highest value
50std::atomic_int64_t minimum= 0; // Lowest value after a maximum
51
52//----------------------------------------------------------------------------
53// pub::Statistic::Methods
54//----------------------------------------------------------------------------
55int64_t // Current value
56 inc( void ) // Increment value
57{
58 counter++; // Increment the counter
59
60 // Increment the current value
61 int64_t old_value= current.load(); // The current value
62 int64_t new_value= old_value + 1;
63 while( !current.compare_exchange_weak(old_value, new_value) )
64 new_value= old_value + 1;
65
66 // Update the maximum and minimum values (if they've changed)
67 old_value= maximum.load();
68 while( old_value < new_value ) { // If necessary, update the maximum
69 if( maximum.compare_exchange_strong(old_value, new_value) ) {
70 // The maximum was successfully updated. Update the minimum.
71 // Note that if some other thread has also updated the maximum and
72 // then also updates the minimum, it's not lowered here.
73 old_value= minimum.load();
74 while( old_value < new_value ) {
75 if( minimum.compare_exchange_strong(old_value, new_value) )
76 break;
77 }
78 break;
79 }
80 }
81
82 return new_value;
83}
84
85int64_t // Current value
86 dec( void ) // Increment value
87{
88 // Decrement the current value
89 int64_t old_value= current.load(); // The current value
90 int64_t new_value= old_value - 1;
91 while( !current.compare_exchange_weak(old_value, new_value) )
92 new_value= old_value - 1;
93
94 // Decrement the minimum value (if it's changed)
95 old_value= minimum.load();
96 while( old_value > new_value ) { // If necessary, update the minimum
97 if( minimum.compare_exchange_strong(old_value, new_value) )
98 break;
99 }
100
101 return new_value;
102}
103}; // struct Statistic
104} // namespace _PUB_NAMESPACE
105#endif // _PUB_STATISTIC_H_INCLUDED
