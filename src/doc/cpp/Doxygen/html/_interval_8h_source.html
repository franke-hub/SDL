42// Interval::Attributes
43//----------------------------------------------------------------------------
44protected:
45std::chrono::steady_clock::time_point
46 start_time; // Start time
47std::chrono::steady_clock::time_point
48 stop_time; // Stop time
49
50//----------------------------------------------------------------------------
51// Interval::Constructors
52//----------------------------------------------------------------------------
53public:
54 ~Interval( void ) {} // Destructor
55 Interval( void ) // Constructor
56: start_time(), stop_time()
57{ start(); }
58
59//----------------------------------------------------------------------------
60// Interval::Operators
61//----------------------------------------------------------------------------
62 operator double( void ) const // (double) cast operator
63{ return to_double(); }
64
65//----------------------------------------------------------------------------
66// Interval::Methods
67//----------------------------------------------------------------------------
68public:
69void
70 start( void ) // Start the Interval Timer
71{
72 start_time= std::chrono::steady_clock::now();
73 stop_time= start_time;
74}
75
76double // Current time difference (seconds)
77 stop( void ) // Stop the interval timer
78{
79 stop_time= std::chrono::steady_clock::now();
80 return to_double();
81}
82
83double // Resultant (in seconds)
84 to_double( void ) const // Convert time difference to double
85{
86 using Nanoseconds= std::chrono::nanoseconds;
87 int64_t ns=
88 std::chrono::duration_cast<Nanoseconds>(stop_time-start_time).count();
89 return (double)ns / 1000000000.0;
90}
91}; // class Interval
