28namespace _PUB_NAMESPACE::dispatch {
29//----------------------------------------------------------------------------
30//
31// Class-
32// Item
33//
34// Purpose-
35// The Dispatcher work Item Object.
36//
37// Implementation notes-
38// The negative function codes are handled internally by the Dispatch
39// object. They are not passed to DispatchTask::work().
40//
41// When post() is invoked:
42// if done == nullptr, the Item is deleted.
43// if done != nullptr, done->done(this) is invoked.
44//
45//----------------------------------------------------------------------------
46class Item : public AI_list<Item>::Link { // A dispatcher work item
47//----------------------------------------------------------------------------
48// Item::Enumerations and typedefs
49//----------------------------------------------------------------------------
50public:
51enum CC // Completion codes
52{ CC_NORMAL= 0 // Normal (OK)
53, CC_ERROR // Generic error
54, CC_PURGE // Function purged
55, CC_INVALID_FC // Function code rejected
56}; // enum CC
57
58enum FC // Function codes
59{ FC_CHASE= (-1) // Chase (NOP)
60, FC_TRACE= (-2) // Trace (NOP)
61#if USE_FC_RESET
62, FC_RESET= (-3) // Reset the Worker
63#else
64#endif
65, FC_VALID= 0 // All user function codes are positive
66}; // enum FC
67
68//----------------------------------------------------------------------------
69// Item::Attributes
70//----------------------------------------------------------------------------
71public:
72int fc; // Function code
73int cc; // Completion code
74Done* done; // Completion callback
75void* work; // (Available for application usage)
76
77//----------------------------------------------------------------------------
78// Item::Constructors
79//----------------------------------------------------------------------------
80public:
81virtual
82 ~Item( void ) { _term(); } // Destructor
83 Item( void ) // Default constructor
84: fc(FC_VALID), cc(CC_NORMAL), done(nullptr) { _init(); }
85
86 Item( // Constructor
87 int fc, // Function code
88 Done* done= nullptr) // -> Done callback object
89: fc(fc), cc(CC_NORMAL), done(done) { _init(); }
90
91private:
92 void _init() const noexcept;
93 void _term() const noexcept;
94
95 Item(const Item&) = delete; // Disallowed copy constructor
96 Item& operator=(const Item&) = delete; // Disallowed assignment operator
97
98//----------------------------------------------------------------------------
99// Item::Methods
100//----------------------------------------------------------------------------
101public:
102virtual void
103 debug(const char* info) const; // Debugging display
104
105void debug( void ) { debug(""); }
106
107void
108 post( // Complete the Work Item
109 int cc= 0) // With this completion code
110{
111 if( done == nullptr )
112 delete this;
113 else {
114 this->cc= cc;
115 done->done(this);
116 }
117}
118}; // class Item
119} // namespace _PUB_NAMESPACE::dispatch
120#endif // _PUB_DISPATCHITEM_H_INCLUDED
