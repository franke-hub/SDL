26#undef TRY_CATCH // (Allow redefine)
27#define TRY_CATCH(x) try { x \
28 } catch(pub::Exception& X) { \
29 fprintf(stderr, "%s\n", std::string(X).c_str()); \
30 } catch(std::exception& X) { \
31 fprintf(stderr, "std::exception.what(%s))\n", X.what()); \
32 } catch(const char* X) { \
33 fprintf(stderr, "catch(const char* '%s')\n", X); \
34 } catch(...) { \
35 fprintf(stderr, "catch(...)\n"); \
36 }
